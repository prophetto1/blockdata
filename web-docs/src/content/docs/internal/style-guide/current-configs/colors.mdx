export const S = ({ c }) => (
  <span style={{ display: 'inline-block', width: '14px', height: '14px', backgroundColor: c, borderRadius: '3px', border: '1px solid rgba(128,128,128,0.35)', verticalAlign: 'middle', marginRight: '6px' }} />

## Dark Theme (default)

### Core Tokens

| Token | Value | Usage |
|-------|-------|-------|
| --background | <S c="#0e0e0e" /> #0e0e0e | Page background |
| --foreground | <S c="#eeeeee" /> #eeeeee | Primary text |
| --primary | <S c="#eb5e41" /> #eb5e41 | Accent / links / interactive |
| --primary-foreground | <S c="#ffffff" /> #ffffff | Text on primary |

### Surface Tokens

| Token | Value | Usage |
|-------|-------|-------|
| --card | <S c="#141414" /> #141414 | Card / panel backgrounds |
| --card-foreground | <S c="#eeeeee" /> #eeeeee | Text on cards |
| --secondary | <S c="#1a1a1a" /> #1a1a1a | Secondary surfaces |
| --secondary-foreground | <S c="#eeeeee" /> #eeeeee | Text on secondary |
| --muted | <S c="#1a1a1a" /> #1a1a1a | Muted backgrounds |
| --muted-foreground | <S c="#a0a0a0" /> #a0a0a0 | De-emphasized text |
| --accent | <S c="#1a1a1a" /> #1a1a1a | Accent backgrounds |
| --accent-foreground | <S c="#eeeeee" /> #eeeeee | Text on accent |

### Border & Input Tokens

| Token | Value | Usage |
|-------|-------|-------|
| --border | <S c="#2a2a2a" /> #2a2a2a | Borders, dividers |
| --input | <S c="#2a2a2a" /> #2a2a2a | Input field borders |
| --ring | <S c="#eb5e41" /> #eb5e41 | Focus rings |
